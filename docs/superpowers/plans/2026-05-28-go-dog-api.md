# Go Dog API Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local Go HTTP server that replicates `dogapi.dog` breed/group endpoints, and wire the Dog-Exploder app to use a configurable base URL read from `appsettings.json`.

**Architecture:** The Go server reads static JSON data files on startup and serves them with runtime pagination — zero external dependencies. The app gains an `AppConfig` static class that reads `appsettings.json` once; both service classes replace hardcoded URLs with `AppConfig` lookups.

**Tech Stack:** Go 1.21 stdlib (`net/http`, `encoding/json`, `math`, `os`), .NET 10 `System.Text.Json`

func TestPaginate_exactOnePage(t *testing.T) {
	data := makeItems(10)
	slice, last := paginate(data, 1)
	if len(slice) != 10 {
		t.Errorf("want 10, got %d", len(slice))
	}

		t.Errorf("want last=1, got %d", last)
	}
}

		t.Errorf("want 0, got %d", len(slice))
	}
	if last != 1 {
		t.Errorf("want last=1, got %d", last)

func TestBreedsHandler_200AndJSON(t *testing.T) {
	breeds = []json.RawMessage{json.RawMessage(`{"id":"1","type":"breed"}`)}

	req := httptest.NewRequest(http.MethodGet, "/api/v2/breeds?page[number]=1", nil)
	w := httptest.NewRecorder()
	handleBreeds(w, req)

	if w.Code != http.StatusOK {
		t.Fatalf("want 200, got %d", w.Code)
	}
	if ct := w.Header().Get("Content-Type"); ct != "application/json; charset=utf-8" {
		t.Errorf("want application/json, got %s", ct)
	}
	var resp apiResponse
	if err := json.NewDecoder(w.Body).Decode(&resp); err != nil {
		t.Fatalf("decode: %v", err)
	}
	if len(resp.Data) != 1 {
		t.Errorf("want 1 item, got %d", len(resp.Data))
	}
	if resp.Meta.Pagination.Current != 1 {
		t.Errorf("want current=1, got %d", resp.Meta.Pagination.Current)
	}
	if resp.Meta.Pagination.Last != 1 {
		t.Errorf("want last=1, got %d", resp.Meta.Pagination.Last)
	}
}

package main

import (
	"encoding/json"
	"log"
	"math"
	"net/http"
	"os"
	"strconv"
)

const pageSize = 10

var breeds []json.RawMessage
var groups []json.RawMessage

func main() {
	breeds = loadJSON("data/breeds.json")
	groups = loadJSON("data/groups.json")

	port := os.Getenv("PORT")
	if port == "" {
		port = "8080"
	}

	mux := http.NewServeMux()
	mux.HandleFunc("/api/v2/breeds", handleBreeds)
	mux.HandleFunc("/api/v2/groups", handleGroups)

	log.Printf("go-dog-api listening on :%s", port)
	log.Fatal(http.ListenAndServe(":"+port, mux))
}

func loadJSON(path string) []json.RawMessage {
	b, err := os.ReadFile(path)
	if err != nil {
		log.Fatalf("load %s: %v", path, err)
	}
	var items []json.RawMessage
	if err := json.Unmarshal(b, &items); err != nil {
		log.Fatalf("parse %s: %v", path, err)
	}
	return items
}

type paginationMeta struct {
	Current int `json:"current"`
	Last    int `json:"last"`
}

type metaWrapper struct {
	Pagination paginationMeta `json:"pagination"`
}

type apiResponse struct {
	Data []json.RawMessage `json:"data"`
	Meta metaWrapper       `json:"meta"`
}

func paginate(data []json.RawMessage, page int) ([]json.RawMessage, int) {
	total := len(data)
	if total == 0 {
		return []json.RawMessage{}, 1
	}
	last := int(math.Ceil(float64(total) / float64(pageSize)))
	if page < 1 {
		page = 1
	}
	if page > last {
		page = last
	}
	start := (page - 1) * pageSize
	end := start + pageSize
	if end > total {
		end = total
	}
	return data[start:end], last
}

func writeJSON(w http.ResponseWriter, data []json.RawMessage, page int) {
	slice, last := paginate(data, page)
	resp := apiResponse{
		Data: slice,
		Meta: metaWrapper{Pagination: paginationMeta{Current: page, Last: last}},
	}
	w.Header().Set("Content-Type", "application/json; charset=utf-8")
	json.NewEncoder(w).Encode(resp)
}

func pageParam(r *http.Request) int {
	n, err := strconv.Atoi(r.URL.Query().Get("page[number]"))
	if err != nil || n < 1 {
		return 1
	}
	return n
}

func handleBreeds(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodGet {
		http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
		return
	}
	writeJSON(w, breeds, pageParam(r))
}

func handleGroups(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodGet {
		http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
		return
	}
	writeJSON(w, groups, pageParam(r))
}

using System.Text.Json;
using System.Text.Json.Serialization;

namespace Dog_Exploder;

internal static class AppConfig
{
    private static readonly Lazy<ConfigData> _data = new(Load, LazyThreadSafetyMode.ExecutionAndPublication);

    public static string DogApiBaseUrl => _data.Value.DogApiBaseUrl;
    public static string DogImageApiBaseUrl => _data.Value.DogImageApiBaseUrl;

    private static ConfigData Load()
    {
        var path = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "appsettings.json");
        if (!File.Exists(path))
            return new ConfigData();
        try
        {
            var json = File.ReadAllText(path);
            return JsonSerializer.Deserialize<ConfigData>(json) ?? new ConfigData();
        }
        catch
        {
            return new ConfigData();
        }
    }

    private sealed class ConfigData
    {
        [JsonPropertyName("DogApiBaseUrl")]
        public string DogApiBaseUrl { get; set; } = "https://dogapi.dog/";

        [JsonPropertyName("DogImageApiBaseUrl")]
        public string DogImageApiBaseUrl { get; set; } = "https://dog.ceo/";
    }
}
